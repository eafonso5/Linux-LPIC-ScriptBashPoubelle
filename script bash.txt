#!/bin/bash

# ==============================================================================
# SCRIPT : trash_manager.sh
# DESCRIPTION : Archive des fichiers dans une "poubelle" et propose la suppression.
# ==============================================================================

TRASH_DIR="/home/$USER/trash"
DATE_FORMAT=$(date +%d_%m_%Y_%H_%M)
ARCHIVE_NAME="archive_$DATE_FORMAT.tar.gz"

# --- Fonction d'usage ---
usage() {
    echo "Usage: $0 fichier1 [fichier2 ... fichier10]"
    echo "Le script accepte entre 1 et 10 fichiers existants."
    exit 1
}

# --- 1. Création des 10 fichiers de test ---
create_test_files() {
    echo "[*] Création des fichiers fichier1 à fichier10..."
    for i in {1..10}; do
        touch "fichier$i"
    done
}

# --- 2. Vérifications initiales ---
# Créer le répertoire trash s'il n'existe pas
if [ ! -d "$TRASH_DIR" ]; then
    mkdir -p "$TRASH_DIR"
fi

# Appeler la création des fichiers (Etape 2 du descriptif)
create_test_files

# Vérifier le nombre d'arguments (Etape 3)
if [ "$#" -lt 1 ] || [ "$#" -gt 10 ]; then
    echo "Erreur : Nombre d'arguments invalide ($#)."
    usage
fi

# Vérifier l'existence de chaque fichier passé en argument
for file in "$@"; do
    if [ ! -f "$file" ]; then
        echo "Erreur : Le fichier '$file' n'existe pas."
        exit 1
    fi
done

# --- 3. Archivage (Etape 4 & 5) ---
echo "[*] Archivage des fichiers vers $TRASH_DIR/$ARCHIVE_NAME..."
tar -czf "$TRASH_DIR/$ARCHIVE_NAME" "$@" 2>/dev/null

if [ $? -eq 0 ]; then
    echo "Succès : Archive créée."
else
    echo "Erreur lors de la création de l'archive."
    exit 1
fi

# --- 4. Demande de suppression (Etape 6) ---
read -p "Voulez-vous supprimer les fichiers originaux ? (o/n) : " confirm

if [[ "$confirm" =~ ^[oO]$ ]]; then
    rm -f "$@"
    echo "[+] Fichiers originaux supprimés."
else
    echo "[-] Les fichiers originaux ont été conservés."
fi

exit 0
